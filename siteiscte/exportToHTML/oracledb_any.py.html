from django.core.exceptions import ImproperlyConfigured
from django.utils.deprecation import RemovedInDjango60Warning

    import oracledb

    is_oracledb = True

        import cx_Oracle as oracledb  # NOQA

        warnings.warn(

            RemovedInDjango60Warning,
